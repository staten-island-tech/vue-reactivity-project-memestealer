<template>
<div class="dashboard">
    <h2>NYPD Arrest Data Explorer</h2>

    <div class="controls">
      <button :class="{ active: currentChart === 'boro' }" @click="currentChart = 'boro'">
        By Borough
      </button>
      <button :class="{ active: currentChart === 'race' }" @click="currentChart = 'race'">
        By Race
      </button>
      <button :class="{ active: currentChart === 'offense' }" @click="currentChart = 'offense'">
        Top Offenses
      </button>
    </div>

    <div class="main-chart-container" v-if="arrests.length">
      <div v-if="currentChart === 'boro'">
        <h3>Arrest Distribution: Borough</h3>
        <Pie :data="boroData" :options="{ responsive: true }" />
      </div>

      <div v-else-if="currentChart === 'race'">
        <h3>Arrest Distribution: Race</h3>
        <Doughnut :data="raceData" :options="{ responsive: true }" />
      </div>

      <div v-else-if="currentChart === 'offense'">
        <h3>Top 5 Offense Categories</h3>
        <Bar :data="offenseData" :options="{ responsive: true }" />
      </div>
    </div>

    <div v-else class="loading">Loading data...</div>

    <hr />

    <div class="container">
      <NYPD v-for="crime in arrests" :key="crime.arrest_key" :arrests="crime" />
    </div>
  </div>
</template>
</template>

<script setup>
import { ref, computed, onMounted } from 'vue'
import { Bar, Pie, Doughnut } from 'vue-chartjs'
import {
  Chart as ChartJS,
  Title,
  Tooltip,
  Legend,
  BarElement,
  CategoryScale,
  LinearScale,
  ArcElement,
} from 'chart.js'

ChartJS.register(Title, Tooltip, Legend, BarElement, CategoryScale, LinearScale, ArcElement)

const arrests = ref([])
const currentChart = ref('boro')

const boroData = computed(() => {
  const boroNames = {
    B: 'Bronx',
    S: 'Staten Island',
    K: 'Brooklyn',
    M: 'Manhattan',
    Q: 'Queens',
  }

  const counts = arrests.value.reduce((acc, arrest) => {
    const code = arrest.arrest_boro
    if (code) {
      const name = boroNames[code] || code
      acc[name] = (acc[name] || 0) + 1
    }
    return acc
  }, {})

  return {
    labels: Object.keys(counts),
    datasets: [
      {
        label: 'Arrests by Borough',
        data: Object.values(counts),
        backgroundColor: ['#FF6384', '#36A2EB', '#FFCE56', '#4BC0C0', '#9966FF'],
      },
    ],
  }
})
const raceData = computed(() => {
  const counts = arrests.value.reduce((acc, arrest) => {
    acc[arrest.perp_race] = (acc[arrest.perp_race] || 0) + 1
    return acc
  }, {})
  return {
    labels: Object.keys(counts),
    datasets: [
      {
        data: Object.values(counts),
        backgroundColor: ['#FF6384', '#000000', '#FFCE56', '#4BC0C0', '#9966FF'],
      },
    ],
  }
})
const offenseData = computed(() => {
  const counts = arrests.value.reduce((acc, arrest) => {
    const offense = arrest.ofns_desc || 'UNKNOWN'

    acc[offense] = (acc[offense] || 0) + 1
    return acc
  }, {})

  const sorted = Object.entries(counts)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 5)

  return {
    labels: sorted.map(([key]) => key),
    datasets: [
      {
        label: 'Number of Arrests',
        data: sorted.map(([, value]) => value),
        backgroundColor: ['#FF6384', '#36A2EB', '#FFCE56', '#4BC0C0', '#9966FF'],
      },
    ],
  }
})
import NYPD from '@/components/nypd.vue'
const arrests = ref([])

async function getArrests() {
  try {
    const response = await fetch('https://data.cityofnewyork.us/resource/uip8-fykc.json')
    const data = await response.json()
    arrests.value = data
  } catch (error) {
    console.log(error)
  }
}
onMounted(() => {
  getArrests()
})
</script>
